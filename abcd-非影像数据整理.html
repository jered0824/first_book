8.3.3 UPPS 衡量冲动行为的一种自我报告问卷

紧迫性（Urgency）、预谋（Premeditation）、毅力（Perseverance）、求刺激（Sensation Seeking）和积极紧迫性（Positive Urgency） 评估个体在不同情境下的冲动倾向以及行为控制能力

# Summary Score(s)    Baseline 2-year follow-up 4-year follow-up
mhy_upps_score <-  fread(here(wd,"datas/5.1/core/mental-health/mh_y_upps.csv"))
names(mhy_upps_score)
# 统计每一项缺失值 
mhy_upps_score %>% 
  summarise(missing_sum = across(ends_with("_nm"), ~sum(.x, na.rm = T))) ->
  upps_miss_sum

mhy_upps_score %>% 
  drop_na() %>% # 去掉缺失值
  group_by(src_subject_id) %>% # 按被试号分组
  filter(n() == 3) %>% # 保留有三次数据的
  filter(src_subject_id %in% sub_id$src_subject_id) %>% # 2463选择有3期Idp的人
  select(src_subject_id,eventname,!ends_with(c("_y","_nm","_nt"))) %>% # 选择需要的列
  ungroup() %>% # 取消分组
  mutate(
    across(where(is.numeric), .fns = list(std = scale), .names = "{.fn}_{.col}")) %>% # 标准化
  left_join(fread(here(wd,"temp/4年随访皮层厚度.csv")),
            by=c("src_subject_id","eventname")) %>% # 与皮层厚度数据合并
  mutate(age = scale(interview_age/12),
    across(.cols = c(src_subject_id,sex), 
           .fns = list(id = ~ as.integer(factor(.x,levels = unique(.x)))),
           .names = "{.fn}_{.col}")
    ) %>% # 性别和被试号转换数值型 年龄标准化
  relocate(c(age,sex,id_src_subject_id,id_sex),  
           .after = eventname) %>% # 调整列的位置
  fwrite(here(wd,"temp/mhy_upps_score.csv")) # 写入文件
